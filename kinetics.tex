\begin{frame}
\frametitle{Reactivity}
		     \begin{align*}
		     k &= \mbox{"neutron multiplication factor"}\\
		     &= \frac{\mbox{neutrons causing fission}}{\mbox{neutrons produced by fission}}\\
		     \rho &= \frac{k-1}{k}\\
		     \rho &= \mbox{reactivity}\\
		     \end{align*}
\end{frame}
